% If microtype is present (loaded by the template), disable it when
% running with luatexja to avoid numerous compatibility warnings.
% Use a macro-existence test because \IfFileExists tests the file system
% and microtype.sty may exist without the package being loaded; calling
% \microtypesetup then would raise "Undefined control sequence".
\makeatletter
\@ifundefined{microtypesetup}{}{\microtypesetup{activate=false, verbose=false}}
\makeatother
